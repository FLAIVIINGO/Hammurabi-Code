Move Zeros Algorithm

Andrew Babilonia

Java

                                
public void moveZeros(int[] nums) {
    int searchIdx = 0;
    for(int i = 0 i < nums.length; i++) {
        if(nums[i] != 0) {
            nums[searchIdx++];
        }
    }
    for(int j = searchIdx; j < nums.length; j++) {
        nums[j] = 0;
    }
}
